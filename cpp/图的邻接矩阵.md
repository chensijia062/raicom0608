//图
//邻接矩阵 
#include<bits/stdc++.h>
using namespace std;

int vd[1005]={0};
int n,m;
int g[1005][1005]={0};
void dfs(int x){
	cout<<x<<" ";
	vd[x]=1;
	for(int i=1;i<=n;i++){
		if(g[x][i]&&vd[i]==0){
			dfs(i);
		}
	}
}
void bfs(int x){
	queue<int> q;
	q.push(x);
	vd[x]=1;
	while(q.size()){
		int temp=q.front();
		q.pop();
		cout<<temp<<" ";
		for(int i=1;i<=n;i++){
			if(g[temp][i]&&vd[i]==0){
				q.push(i);
				vd[i]=1;
			}
		}
	}
}

int main()
{
	
	cin>>n>>m;
	for(int i=0;i<m;i++){
		int u,v;
		cin>>u>>v;
		g[u][v]=g[v][u]=1;
	}
	//输出邻接矩阵
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			cout<<g[i][j]<<" ";
		}
		cout<<endl;
	} 
	//DFS
	dfs(1);
	//BFS
	bfs(1);
	return 0;
 }
