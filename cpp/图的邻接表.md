//图
//邻接表
#include<bits/stdc++.h>
using namespace std;

int vd[1005]={0};
vector<int> edge[1005];
void dfs(int x){
    vd[x]=1;
    cout<<x<<" ";
    for(int i=0;i<edge[x].size();i++)
    {
        int v=edge[x][i];
        if(vd[v]==0)
        {
            dfs(v);
        }
    }
}
void bfs(int x){
	queue<int> q;
	q.push(x);
	vd[x]=1;
	while(q.size()){
		int temp=q.front();
		q.pop();
		cout<<temp<<" ";
		for(int i=0;i<edge[temp].size();i++){
			if(vd[edge[temp][i]]==0){
				q.push(edge[temp][i]);
				vd[edge[temp][i]]=1;
			}
		}
	}
	return;
}
int main()
{
	int n,m;
	cin>>n>>m;
	
	for(int i=0;i<m;i++){
		int u,v;
		cin>>u>>v;
		edge[u].push_back(v);
		edge[v].push_back(u);
	} 
	//输出邻接表 
	for(int i=1;i<=n;i++){
		cout<<i<<":";
		for(int j=0;j<edge[i].size();j++){
			cout<<edge[i][j]<<" ";
		}
		cout<<endl;
	}
	//DFS
	dfs(1);
	//BFS
	bfs(1);
	return 0;
 }
